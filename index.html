Práctica Guiada:

Crear programa Arduino que lea temperatura y humedad del sensor DHT22 cada 2 segundos y envíe los datos por puerto serial en formato CSV.

// Código Arduino - Lectura DHT22 y envío serial
#include <DHT.h>

#define DHTPIN 2
#define DHTTYPE DHT22
DHT dht(DHTPIN, DHTTYPE);

void setup() {
  Serial.begin(9600);
  dht.begin();
  delay(2000); // Estabilización sensor
}

void loop() {
  float temp = dht.readTemperature();
  float hum = dht.readHumidity();
  
  if (!isnan(temp) && !isnan(hum)) {
    // Formato: timestamp,temperatura,humedad
    Serial.print(millis());
    Serial.print(",");
    Serial.print(temp, 2);
    Serial.print(",");
    Serial.println(hum, 2);
  }
  
  delay(2000); // Lectura cada 2 segundos
}

Práctica Guiada:

Desarrollar script Python que capture datos de Arduino, aplique limpieza (detección de outliers con método IQR), y almacene en archivo CSV limpio.

# Script Python - Captura y limpieza de datos
import serial
import pandas as pd
from datetime import datetime
import numpy as np

# Conexión serial
arduino = serial.Serial('COM3', 9600, timeout=1)
datos = []

print("Iniciando captura de datos... (Ctrl+C para detener)")

try:
    while True:
        if arduino.in_waiting > 0:
            linea = arduino.readline().decode('utf-8').strip()
            valores = linea.split(',')
            
            try:
                registro = {
                    'timestamp': datetime.now(),
                    'tiempo_ms': int(valores[0]),
                    'temperatura': float(valores[1]),
                    'humedad': float(valores[2])
                }
                
                # Validación básica
                if 0 <= registro['temperatura'] <= 50 and 0 <= registro['humedad'] <= 100:
                    datos.append(registro)
                    print(f"✓ T: {registro['temperatura']}°C | H: {registro['humedad']}%")
                else:
                    print("⚠ Dato fuera de rango - descartado")
                    
            except (ValueError, IndexError) as e:
                print(f"⚠ Error al parsear: {e}")
                
except KeyboardInterrupt:
    # Guardar y limpiar datos
    df = pd.DataFrame(datos)
    
    # Detección de outliers (IQR)
    Q1 = df['temperatura'].quantile(0.25)
    Q3 = df['temperatura'].quantile(0.75)
    IQR = Q3 - Q1
    df_limpio = df[~((df['temperatura'] < (Q1 - 1.5 * IQR)) | 
                     (df['temperatura'] > (Q3 + 1.5 * IQR)))]
    
    print(f"\nRegistros capturados: {len(df)}")
    print(f"Outliers eliminados: {len(df) - len(df_limpio)}")
    print(f"Registros limpios: {len(df_limpio)}")
    
    df_limpio.to_csv('datos_limpios.csv', index=False)
    print("\n✓ Datos guardados en 'datos_limpios.csv'")
    
    arduino.close()

-- Script SQL - Creación de base de datos
CREATE DATABASE analitica_industrial;

-- Tabla de sensores
CREATE TABLE sensores (
    id_sensor SERIAL PRIMARY KEY,
    nombre VARCHAR(50) NOT NULL UNIQUE,
    tipo VARCHAR(30) NOT NULL,
    ubicacion VARCHAR(100),
    unidad_medida VARCHAR(20),
    fecha_instalacion DATE DEFAULT CURRENT_DATE,
    estado VARCHAR(20) DEFAULT 'activo'
);

-- Tabla de lecturas
CREATE TABLE lecturas (
    id_lectura SERIAL PRIMARY KEY,
    id_sensor INTEGER NOT NULL REFERENCES sensores(id_sensor),
    timestamp TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    valor NUMERIC(10,2) NOT NULL,
    calidad VARCHAR(20) DEFAULT 'buena',
    CONSTRAINT valor_valido CHECK (valor >= -50 AND valor <= 150)
);

-- Índices para optimización
CREATE INDEX idx_lecturas_timestamp ON lecturas(timestamp DESC);
CREATE INDEX idx_lecturas_sensor_tiempo ON lecturas(id_sensor, timestamp DESC);

-- Insertar sensores de ejemplo
INSERT INTO sensores (nombre, tipo, ubicacion, unidad_medida) VALUES
    ('Sensor_Temp_01', 'Temperatura', 'Línea de Producción A', '°C'),
    ('Sensor_Hum_01', 'Humedad', 'Línea de Producción A', '%');
# Script Python - Integración completa Arduino-PostgreSQL
import serial
import psycopg2
from datetime import datetime

# Conexión a PostgreSQL
try:
    conn = psycopg2.connect(
        host="localhost",
        database="analitica_industrial",
        user="postgres",
        password="tu_password"
    )
    cur = conn.cursor()
    print("✓ Conectado a PostgreSQL")
except Exception as e:
    print(f"✗ Error de conexión: {e}")
    exit()

# Conexión a Arduino
arduino = serial.Serial('COM3', 9600, timeout=1)
print("✓ Conectado a Arduino")

def insertar_lectura(id_sensor, valor):
    try:
        query = """
        INSERT INTO lecturas (id_sensor, timestamp, valor) 
        VALUES (%s, %s, %s)
        """
        cur.execute(query, (id_sensor, datetime.now(), valor))
        conn.commit()
        return True
    except Exception as e:
        print(f"✗ Error inserción: {e}")
        conn.rollback()
        return False

print("\nIniciando pipeline Arduino → PostgreSQL...")
print("Presiona Ctrl+C para detener\n")

contador = 0
try:
    while True:
        if arduino.in_waiting > 0:
            linea = arduino.readline().decode('utf-8').strip()
            valores = linea.split(',')
            
            try:
                temp = float(valores[1])
                hum = float(valores[2])
                
                # Insertar temperatura (sensor 1)
                if insertar_lectura(1, temp):
                    contador += 1
                    
                # Insertar humedad (sensor 2)
                if insertar_lectura(2, hum):
                    contador += 1
                    
                print(f"✓ [{contador}] T: {temp}°C | H: {hum}% → BD")
                
            except (ValueError, IndexError) as e:
                print(f"⚠ Error al parsear: {e}")
                
except KeyboardInterrupt:
    print(f"\n\n✓ Pipeline detenido. {contador} registros insertados.")
    cur.close()
    conn.close()
    arduino.close()

-- Consultas SQL Analíticas

-- Vista: Promedios por hora
CREATE VIEW v_promedios_hora AS
SELECT 
    DATE_TRUNC('hour', l.timestamp) as hora,
    s.nombre as sensor,
    s.tipo,
    AVG(l.valor) as valor_promedio,
    MIN(l.valor) as valor_minimo,
    MAX(l.valor) as valor_maximo,
    COUNT(*) as num_lecturas
FROM lecturas l
JOIN sensores s ON l.id_sensor = s.id_sensor
GROUP BY DATE_TRUNC('hour', l.timestamp), s.nombre, s.tipo
ORDER BY hora DESC;

-- Vista: Últimas 24 horas
CREATE VIEW v_ultimas_24h AS
SELECT 
    s.nombre,
    s.tipo,
    l.timestamp,
    l.valor,
    s.unidad_medida
FROM lecturas l
JOIN sensores s ON l.id_sensor = s.id_sensor
WHERE l.timestamp >= NOW() - INTERVAL '24 hours'
ORDER BY l.timestamp DESC;

-- Vista: KPIs actuales
CREATE VIEW v_kpis_actuales AS
SELECT 
    s.nombre,
    s.tipo,
    ROUND(AVG(l.valor)::numeric, 2) as valor_actual,
    s.unidad_medida,
    MAX(l.timestamp) as ultima_lectura
FROM lecturas l
JOIN sensores s ON l.id_sensor = s.id_sensor
WHERE l.timestamp >= NOW() - INTERVAL '5 minutes'
GROUP BY s.nombre, s.tipo, s.unidad_medida;

/* Medidas DAX para el Dashboard */

// Temperatura Actual (último valor)
Temp_Actual = 
VAR UltimaFecha = MAX(lecturas[timestamp])
RETURN
    CALCULATE(
        AVERAGE(lecturas[valor]),
        lecturas[timestamp] = UltimaFecha,
        RELATED(sensores[tipo]) = "Temperatura"
    )

// Temperatura Promedio Última Hora
Temp_UltimaHora = 
CALCULATE(
    AVERAGE(lecturas[valor]),
    DATESINPERIOD(lecturas[timestamp], MAX(lecturas[timestamp]), -1, HOUR),
    RELATED(sensores[tipo]) = "Temperatura"
)

// Variación vs Hora Anterior
Variacion_Horaria = 
VAR TempActual = [Temp_UltimaHora]
VAR TempAnterior = 
    CALCULATE(
        [Temp_UltimaHora],
        DATEADD(lecturas[timestamp], -1, HOUR)
    )
RETURN
    TempActual - TempAnterior

// Porcentaje de Variación
Variacion_Porcentaje = 
DIVIDE([Variacion_Horaria], [Temp_UltimaHora], 0) * 100

// Contador de Alertas (valores fuera de rango)
Alertas = 
CALCULATE(
    COUNTROWS(lecturas),
    lecturas[valor] > 30 || lecturas[valor] < 18
)

// Temperatura Máxima del Día
Temp_Max_Dia = 
CALCULATE(
    MAX(lecturas[valor]),
    TODAY(),
    RELATED(sensores[tipo]) = "Temperatura"
)

# Script Python - Análisis KDD Final
import pandas as pd
import psycopg2
import numpy as np
from scipy import stats

# Conexión y extracción de datos
conn = psycopg2.connect(
    host="localhost",
    database="analitica_industrial",
    user="postgres",
    password="tu_password"
)

query = """
SELECT 
    l.timestamp,
    s.nombre as sensor,
    s.tipo,
    l.valor
FROM lecturas l
JOIN sensores s ON l.id_sensor = s.id_sensor
ORDER BY l.timestamp
"""

df = pd.read_sql(query, conn)
conn.close()

print("=" * 50)
print("ANÁLISIS KDD - DESCUBRIMIENTO DE CONOCIMIENTO")
print("=" * 50)

# 1. Estadísticas descriptivas
print("\n1. ESTADÍSTICAS DESCRIPTIVAS:")
print(df.groupby('tipo')['valor'].describe())

# 2. Detección de outliers
print("\n2. DETECCIÓN DE ANOMALÍAS:")
for tipo in df['tipo'].unique():
    datos_tipo = df[df['tipo'] == tipo]['valor']
    z_scores = np.abs(stats.zscore(datos_tipo))
    outliers = np.sum(z_scores > 3)
    print(f"   {tipo}: {outliers} valores anómalos detectados")

# 3. Análisis de tendencia
print("\n3. TENDENCIAS TEMPORALES:")
df['timestamp'] = pd.to_datetime(df['timestamp'])
df.set_index('timestamp', inplace=True)

for tipo in df['tipo'].unique():
    datos_tipo = df[df['tipo'] == tipo]['valor']
    tendencia = "ASCENDENTE" if datos_tipo.iloc[-10:].mean() > datos_tipo.iloc[:10].mean() else "DESCENDENTE"
    print(f"   {tipo}: Tendencia {tendencia}")

# 4. Correlación entre sensores
print("\n4. CORRELACIÓN ENTRE VARIABLES:")
pivot = df.pivot_table(values='valor', index='timestamp', columns='tipo')
correlacion = pivot.corr()
print(correlacion)

# 5. Patrones por hora del día
print("\n5. PATRONES TEMPORALES:")
df['hora'] = df.index.hour
promedio_hora = df.groupby(['tipo', 'hora'])['valor'].mean()
print(promedio_hora)

print("\n" + "=" * 50)
print("Análisis completado. Revise los insights identificados.")
print("=" * 50)
